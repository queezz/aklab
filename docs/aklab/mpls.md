def ticks_visual(ax, **kwarg):
    &#34;&#34;&#34;
    makes auto minor and major ticks for matplotlib figure
    makes minor and major ticks thicker and longer
    &#34;&#34;&#34;
    which = kwarg.get(&#34;which&#34;, &#34;both&#34;)
    from matplotlib.ticker import AutoMinorLocator

    if which == &#34;both&#34; or which == &#34;x&#34;:
        ax.xaxis.set_minor_locator(AutoMinorLocator())
    if which == &#34;both&#34; or which == &#34;y&#34;:
        ax.yaxis.set_minor_locator(AutoMinorLocator())

    l1 = kwarg.get(&#34;l1&#34;, 7)
    l2 = kwarg.get(&#34;l2&#34;, 4)
    w1 = kwarg.get(&#34;w1&#34;, 1.0)
    w2 = kwarg.get(&#34;w2&#34;, 0.8)
    ax.xaxis.set_tick_params(width=w1, length=l1, which=&#34;major&#34;)
    ax.xaxis.set_tick_params(width=w2, length=l2, which=&#34;minor&#34;)
    ax.yaxis.set_tick_params(width=w1, length=l1, which=&#34;major&#34;)
    ax.yaxis.set_tick_params(width=w2, length=l2, which=&#34;minor&#34;)
    return


def grid_visual(ax, alpha=[0.1, 0.3]):
    &#34;&#34;&#34;
    Sets grid on and adjusts the grid style.
    &#34;&#34;&#34;
    ax.grid(which=&#34;minor&#34;, linestyle=&#34;-&#34;, alpha=alpha[0])
    ax.grid(which=&#34;major&#34;, linestyle=&#34;-&#34;, alpha=alpha[1])
    return


def gritix(**kws):
    &#34;&#34;&#34;

    currently active pylab axes.
    &#34;&#34;&#34;

    ticks_visual(plt.gca())
    grid_visual(plt.gca())
    return

def ltexp(exp, decplace=1, short=False):
    &#34;&#34;&#34; converts 1e29 float to the scientific notation LaTeX string &#34;&#34;&#34;
    import numpy as np

    exponent = int(np.floor(np.log10(abs(exp))))
    coeff = round(exp / np.float64(10 ** exponent), decplace)
    if not short:
        return f&#34;{coeff}\\times 10^{{{exponent}}}&#34;
    else:
        return f&#34;10^{{{exponent}}}&#34;

def set_size(width, fraction=1, subplots=(1, 1), ratio=&#34;golden&#34;):
    &#34;&#34;&#34;Set figure dimensions to avoid scaling in LaTeX.

    Parameters
    ----------
    width: float or string
            Document width in points, or string of predined document type
    fraction: float, optional
            Fraction of the width which you wish the figure to occupy
    subplots: array-like, optional
            The number of rows and columns of subplots.
    ratio: float or string.
            ratio = fig_hight_pt/fig_width_pt. If &#39;golden&#39; equals to golden ratio
    Returns
    -------
    fig_dim: tuple
            Dimensions of figure in inches
    &#34;&#34;&#34;
    if width == &#34;thesis&#34;:
        width_pt = 426.79135
    elif width == &#34;beamer&#34;:
        width_pt = 307.28987
    else:
        width_pt = width

    # Width of figure (in pts)
    fig_width_pt = width_pt * fraction
    # Convert from pt to inches
    inches_per_pt = 1 / 72.27

    # Golden ratio to set aesthetic figure height
    # https://disq.us/p/2940ij3
    golden_ratio = (5 ** 0.5 - 1) / 2

    if ratio == &#34;golden&#34;:
        ratio = golden_ratio

    # Figure width in inches
    fig_width_in = fig_width_pt * inches_per_pt
    # Figure height in inches
    fig_height_in = fig_width_in * ratio * (subplots[0] / subplots[1])

    return (fig_width_in, fig_height_in)
